2. Calculating Distribution Metrics (18-54 Population)

Code 
# 1. Filter to 18-54 and Recalculate 100% Denominators
plot_data_filtered <- plot_data %>%
  filter(year != 2013) %>%
  filter(age_bracket %in% c("18-24", "25-34", "35-44", "45-54")) %>%
  group_by(year, dataset) %>%
  mutate(
    total_pop_year = sum(population_count, na.rm = TRUE),
    percentage = population_count / total_pop_year
  ) %>%
  ungroup()

# 2. Recalculate Metrics (PSI, KS) for the new 18-54 Universe
metrics_calc <- plot_data_filtered %>%
  group_by(year, age_bracket, gender, dataset) %>%
  summarise(percentage = sum(percentage, na.rm = TRUE), .groups = "drop") %>%
  pivot_wider(names_from = dataset, values_from = percentage) %>%
  rename(P_ACS = `Imputed ACS`, Q_DHS = `DHS Official`) %>%
  mutate(
    P_ACS = if_else(is.na(P_ACS) | P_ACS == 0, 0.00001, as.numeric(P_ACS)),
    Q_DHS = if_else(is.na(Q_DHS) | Q_DHS == 0, 0.00001, as.numeric(Q_DHS))
  )

final_metrics <- metrics_calc %>%
  group_by(year) %>%
  summarise(
    PSI = sum((P_ACS - Q_DHS) * log(P_ACS / Q_DHS)),
    .groups = "drop"
  ) %>%
  left_join(
    metrics_calc %>%
      group_by(year, age_bracket) %>%
      summarise(P_ACS_Age = sum(P_ACS), Q_DHS_Age = sum(Q_DHS), .groups = "drop") %>%
      group_by(year) %>%
      arrange(year, age_bracket) %>%
      mutate(CDF_ACS = cumsum(P_ACS_Age), CDF_DHS = cumsum(Q_DHS_Age), abs_diff = abs(CDF_ACS - CDF_DHS)) %>%
      summarise(Max_CDF_Diff_Age = max(abs_diff), .groups = "drop"),
    by = "year"
  )

3. Distribution Shift Metrics Over Time

library(ggplot2)

# Pivot metrics for plotting
metrics_long <- final_metrics %>%
  pivot_longer(
    cols = c(PSI, Max_CDF_Diff_Age),
    names_to = "Metric",
    values_to = "Value"
  ) %>%
  mutate(
    Metric = case_when(
      Metric == "Max_CDF_Diff_Age" ~ "Max CDF Difference (KS Proxy)",
      TRUE ~ Metric
    )
  )

# Plot the trendlines
p_metrics_trend <- ggplot(metrics_long, aes(x = year, y = Value, color = Metric, group = Metric)) +
  geom_line(linewidth = 1.2) +
  geom_point(size = 3) +
  scale_x_continuous(breaks = c(2009:2012, 2014:2019)) +
  scale_color_manual(values = c("Max CDF Difference (KS Proxy)" = "#BA4A00", "PSI" = "#5D3A9B")) +
  theme_minimal(base_size = 14) +
  labs(
    title = "Distribution Shift Metrics Over Time",

library(scales)

# 1. Configure the Scorecards
scorecards <- final_metrics %>%

p_bars_final <- ggplot(plot_data_bars, aes(x = plot_x, y = percentage, fill = dataset)) +

  geom_text(aes(label = percent(percentage, accuracy = 0.1)), 
            vjust = -0.5, size = 3, fontface = "bold") +

  # Recalculate the denominator so these 4 brackets equal exactly 100%
  group_by(year, dataset) %>%
  mutate(
    total_pop_year = sum(population_count, na.rm = TRUE),
    percentage = population_count / total_pop_year
