# --- PRE-REQUISITE: ASSUMED STARTING DATA ---
# acs_base: Your ACS data from the previous step (has year, state_name, bpld, high_prob_count)
# dhs_states: CSV with year, state_name, dhs_count
# dhs_countries: CSV with year, country_name, dhs_count

# --- 1. COUNTRY OF ORIGIN HARMONIZATION ---
# Create a crosswalk for the Top DHS Countries using BPLD codes
country_map <- c(
  "20000" = "Mexico",
  "21010" = "El Salvador",
  "21020" = "Guatemala",
  "21030" = "Honduras",
  "50100" = "India",
  "50000" = "China",
  "51500" = "Philippines"
  # Add the rest of the DHS explicitly named countries here...
)

acs_mapped <- acs_base %>%
  mutate(
    # Translate the code to a name. If it's not in our map, label it "Other"
    country_name = coalesce(country_map[as.character(bpld)], "Other Countries")
  )

# --- 2. STATE DISTRIBUTION COMPARISON ---
# Calculate the percentage of the total population living in each state
state_distribution <- acs_mapped %>%
  # 1. Roll up ACS counts by State
  group_by(year, state_name) %>%
  summarise(imputed_count = sum(high_recall_count, na.rm = TRUE), .groups = "drop") %>%
  # 2. Calculate the ACS State Percentage
  group_by(year) %>%
  mutate(imputed_pct = (imputed_count / sum(imputed_count)) * 100) %>%
  ungroup() %>%
  
  # 3. Join with DHS States
  left_join(dhs_states, by = c("year", "state_name")) %>%
  
  # 4. Calculate the DHS State Percentage & The Variance
  group_by(year) %>%
  mutate(
    dhs_pct = (Count / sum(Count, na.rm = TRUE)) * 100,
    # Positive variance means your model over-allocated to this state vs DHS
    dist_variance = imputed_pct - dhs_pct 
  ) %>%
  ungroup() %>%
  filter(!is.na(state_name))

# --- 3. COUNTRY DISTRIBUTION COMPARISON ---
# Calculate the percentage of the total population originating from each country
country_distribution <- acs_mapped %>%
  # 1. Roll up ACS counts by Country (including the "Other" bucket)
  group_by(year, country_name) %>%
  summarise(imputed_count = sum(high_prob_count, na.rm = TRUE), .groups = "drop") %>%
  # 2. Calculate the ACS Country Percentage
  group_by(year) %>%
  mutate(imputed_pct = (imputed_count / sum(imputed_count)) * 100) %>%
  ungroup() %>%
  
  # 3. Join with DHS Countries
  left_join(dhs_countries, by = c("year", "country_name")) %>%
  
  # 4. Calculate the DHS Country Percentage & The Variance
  group_by(year) %>%
  mutate(
    dhs_pct = (Count / sum(Count, na.rm = TRUE)) * 100,
    dist_variance = imputed_pct - dhs_pct
  ) %>%
  ungroup()
# Example: Visualizing the State Variance for 2019
state_variance_plot <- state_distribution %>%
  filter(year == 2019) %>%
  # Filter to the top 15 states by DHS volume so the chart isn't cluttered
  slice_max(order_by = Count, n = 5) %>%
  mutate(
    # Create a flag for coloring: Over-estimated vs Under-estimated
    variance_direction = ifelse(dist_variance > 0, "Over-Allocated", "Under-Allocated")
  ) %>%
  ggplot(aes(x = reorder(state_name, dist_variance), y = dist_variance, fill = variance_direction)) +
  geom_col() +
  coord_flip() + # Makes it a horizontal bar chart
  scale_fill_manual(values = c("Over-Allocated" = "#2ca02c", "Under-Allocated" = "#d62728")) +
  labs(
    title = "Imputation Variance by State (2019)",
    subtitle = "Percentage point difference from official DHS state distributions",
    x = "",
    y = "Variance (Percentage Points)",
    fill = "Model Behavior vs DHS"
  ) +
  theme_minimal()

ggplotly(state_variance_plot)
