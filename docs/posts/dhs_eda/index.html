The Objective

Before deploying any advanced modeling, it is critical to understand the shape, missingness, and distributions of the raw Department of Homeland Security data.

# --- 2. PLOT 1: THE OVERALL TOTAL ---
# Roll up the data ignoring all demographic categories
total_yearly <- acs_summary %>%
  group_by(year) %>%
  summarise(total_high_recall = sum(high_recall_count, na.rm = TRUE), .groups = 'drop')

# Build the plot using ggplot2
p_total <- ggplot(total_yearly, aes(x = as.character(year), y = total_high_recall)) +
  geom_col(fill = "#2E86C1") +
  labs(
    title = "Total Imputed DHS Records by Year",
    x = "Year",
    y = "Total Imputations"
  ) +
  theme_minimal()

# Convert to an interactive web chart
ggplotly(p_total, tooltip = c("y", "x"))
# --- 3. PLOT 2: THE DEMOGRAPHIC BREAKDOWN ---
age_yearly <- acs_summary %>%
  group_by(year, age_bracket) %>%
  summarise(total_high_recall = sum(high_recall_count, na.rm = TRUE), .groups = 'drop')

# Build a grouped bar chart
p_age <- ggplot(age_yearly, aes(x = as.character(year), y = total_high_recall, fill = age_bracket)) +
  # position = "dodge" puts the bars side-by-side instead of stacking them
  geom_col(position = "dodge", alpha = 0.9) + 
  scale_fill_viridis_d(option = "mako") + # Adds a professional color palette
  labs(
    title = "Imputed Records by Year and Age Bracket",
    x = "Year",
    y = "Total Imputations",
    fill = "Age Bracket"
  ) +
  theme_minimal()

# Convert to an interactive web chart
ggplotly(p_age, tooltip = c("fill", "y", "x"))
