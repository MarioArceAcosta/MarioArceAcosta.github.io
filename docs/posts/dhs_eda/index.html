  summarise(
    total_foreign_born = sum(total_foreign_born_pop, na.rm = TRUE),
    # Let's use your high probability model for this example
    imputed_count = sum(high_prob_count, na.rm = TRUE) 
  )

# Join and Calculate Rates
macro_comparison <- macro_acs %>%
  # Assuming dhs_totals has columns: year, dhs_unauth_count
  left_join(totals, by = "year") %>%
  mutate(
    imputation_rate = imputed_count / total_foreign_born,
    dhs_rate = Count / total_foreign_born,
    variance = imputation_rate - dhs_rate
  )

Age Composition

To assess whether our machine-learning imputations of undocumented status aligned with the estimates provided by DHS (both derived from the ACS to begin with), we examine the age composition of the undocumented population over the years 2009-2019. We can then compare how closely, or how far, the age composition of our machine-learning undocumented population estimates follow the age composition reported by DHS.

library(tidyverse)
library(scales)
library(plotly)

# --- 1. CLEAN DHS STRINGS ---
dhs_age_clean <- dhs_age %>%
  mutate(age_bracket = str_replace_all(age_bracket, '=|"', ''))

# --- 2. PREP YOUR IMPUTED DATA ---
imputed_bracket_dist <- acs_base %>%
  filter(age_bracket != "Under 18") %>%
  group_by(year, age_bracket) %>%
  summarise(imputed_count = sum(high_recall_count, na.rm = TRUE), .groups = "drop") %>%
  group_by(year) %>%
  mutate(
    fraction = imputed_count / sum(imputed_count),
    dataset = "My Imputed ACS"
  ) %>%
  ungroup()

# --- 3. PREP YOUR BENCHMARK DATA ---
dhs_age_benchmark <- dhs_age_clean %>%
   filter(age_bracket != "Under 18") %>%
   group_by(Data_Year, age_bracket) %>%
   summarise(bench_count = sum(Count, na.rm = TRUE), .groups = "drop") %>%
   group_by(Data_Year) %>%
   mutate(
     fraction = bench_count / sum(bench_count),
     dataset = "DHS Official"
   ) %>%
   ungroup() %>%
   rename(year = Data_Year)

# Combine datasets and lock in the chronological factor levels
plot_data <- bind_rows(imputed_bracket_dist, dhs_age_benchmark) %>%
  mutate(
    age_bracket = factor(age_bracket, levels = c("18-24", "25-34", "35-44", "45-54", "55 And Over"))
  ) %>%
  select(year, age_bracket, fraction, dataset)

# --- 4. BUILD THE FACETED PLOT ---
p_age_compare <- ggplot(plot_data, aes(x = age_bracket, y = fraction, fill = dataset)) +
  # Using your exact, stable syntax
  geom_col(position = "dodge", alpha = 0.9) +
  
  scale_fill_manual(values = c("DHS Official" = "#4472C4", "My Imputed ACS" = "#ED7D31")) +
  scale_y_continuous(labels = percent_format(accuracy = 1)) +
  
  labs(
    title = "Age Distribution: Imputed ACS vs. DHS Official",
    x = "Age Bracket",

  ) +
