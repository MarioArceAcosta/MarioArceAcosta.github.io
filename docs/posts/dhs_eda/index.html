The Objective

Before deploying any advanced modeling, it is critical to understand the shape, missingness, and distributions of the raw Department of Homeland Security data. The primary objective for this post is to examine the demographic distribution of the DHS estimates from 2009-2019, specifically focusing on the age and gender composition of the undocumented population over the decade.

Loading the Data

We’ll start by pulling in the summary statistics and estimates from our database and our extracted DHS tables. Data preparation can be described in two steps:

  1. Training machine learning models on cleaned SIPP data to impute undocumented status in the ACS using gradient-boosted trees. This process is identical to what can be found here.
  2. Creating a high-level snapshot of this data to be uploaded via SQL to Supabase for future use, alongside our extracted official DHS demographic tables.

library(tidyverse)
library(DBI)
library(RPostgres)

con <- dbConnect(

  dbname = "postgres",
  host = "aws-1-us-east-2.pooler.supabase.com", 
  port = 6543,
  user = "postgres.dpuhypyqwupzesfkiplu",
  password = Sys.getenv("SUPABASE_PWD")

acs_foreign_summary <- dbGetQuery(con, "SELECT * FROM acs_foreign_summary")
dbDisconnect(con)

# --- 2. LOAD DHS CSVs ---
dhs_male <- read_csv("Unauthorized Immigrant Population Estimates (Male) 2009-2022 - Unauthorized Immigrant Population Estimates (Male) 2009-2022.csv") %>%
  filter(Year >= 2009 & Year <= 2019) %>%
  rename(year = Year, age_bracket = Age, population_count = Count) %>%
  mutate(gender = "Male")

dhs_female <- read_csv("Unauthorized Immigrant Population Estimates (Female) 2009-2022 - Unauthorized Immigrant Population Estimates (Female) 2009-2022.csv") %>%
  filter(Year >= 2009 & Year <= 2019) %>%
  rename(year = Year, age_bracket = Age, population_count = Count) %>%
  mutate(gender = "Female")

dhs_combined <- bind_rows(dhs_male, dhs_female) %>%
  mutate(
    age_bracket = str_trim(str_replace_all(age_bracket, '=|"', '')),
    dataset = "DHS Official"
  ) %>%
  filter(age_bracket != "Under 18", age_bracket != "Total", !is.na(year))

# --- 3. PREP ACS GENDER DATA ---
acs_combined <- acs_foreign_summary %>%
  # Translate ACS sex codes to match DHS text
  mutate(
    gender = case_when(
      fem == 1 ~ "Female",
      fem == 0 ~ "Male",
      TRUE ~ "Unknown"
    )
  ) %>%
  filter(age_bracket != "Under 18", gender != "Unknown", !is.na(year)) %>%
  group_by(year, age_bracket, gender) %>%
  summarise(population_count = sum(high_recall_count, na.rm = TRUE), .groups = "drop") %>%
  mutate(dataset = "Imputed ACS")

# --- 4. MERGE AND APPLY THE OFFSET TRICK ---
plot_data <- bind_rows(dhs_combined, acs_combined) %>%
  mutate(
    age_bracket = factor(age_bracket, levels = c("18-24", "25-34", "35-44", "45-54", "55 And Over")),
    gender = factor(gender, levels = c("Male", "Female")),
    
    # Create the interaction grouping for colors
    fill_group = paste(gender, dataset, sep = " - "),
    
    # THE OFFSET TRICK: Shift ACS slightly left, DHS slightly right
    age_numeric = as.numeric(age_bracket),
    plot_x = ifelse(dataset == "Imputed ACS", age_numeric - 0.2, age_numeric + 0.2)
  )

Age & Gender Composition

To establish a solid demographic baseline, we examine the age and gender composition of the undocumented population reported by DHS over the years 2009-2019. By stacking the male and female estimates, we can simultaneously observe the total size of each age bracket alongside its internal gender distribution.

Faceted/Overview Plots of DHS Demographics

Code 
library(ggplot2)
library(scales) 
library(plotly)

p_comparison_faceted <- ggplot(plot_data, aes(x = plot_x, y = population_count, fill = fill_group)) +
  # Because the X coordinates are physically separated, "stack" will only stack 
  # Male on top of Female within the same dataset!
  geom_col(position = "stack", color = "black", alpha = 0.9, width = 0.35) +
  
  # Assign distinct colors to distinguish ACS from DHS visually
  scale_fill_manual(values = c(
    "Male - DHS Official"   = "#2471A3", # Dark Blue
    "Female - DHS Official" = "#7FB3D5", # Dark Red
    "Male - Imputed ACS"   = "#BA4A00", # Light Blue
    "Female - Imputed ACS" = "#F5B041"  # Light Red
  )) +
  
  # Map the numeric X-axis back to the categorical Age Bracket text
  scale_x_continuous(
    breaks = 1:5,
    labels = levels(plot_data$age_bracket)
  ) +
  scale_y_continuous(labels = label_comma()) +
  
  labs(
    title = "Population Estimates by Age and Gender: Imputed ACS vs DHS Official",
    x = "Age Bracket",
    y = "Total Population Estimate",
    fill = "Dataset & Gender"
  ) +
  
  theme_minimal() +
  theme(
    axis.text.x = element_text(angle = 45, hjust = 1, color = "black"),
    panel.border = element_rect(color = "gray80", fill = NA),
    legend.position = "bottom",
    
    # 2. Force extra space between the facet panels
    panel.spacing = unit(2, "lines"), 
    # Make the year titles a bit bolder to stand out
    strip.text = element_text(face = "bold", margin = margin(b = 10, t = 10)) 
  ) +
  facet_wrap(~ year, ncol = 3)

# 3. Give the ggplotly conversion a slight margin push for its own legend
ggplotly(p_comparison_faceted, tooltip = c("fill", "y")) %>%
  layout(legend = list(orientation = "h", x = 0.5, xanchor = "center", y = -0.1), margin = list(b = 60))

Year-by-Year Interactive Composition

Code 
library(plotly)

p_interactive_comparison <- plot_ly(
  data = plot_data,
  x = ~plot_x,
  y = ~population_count,
  color = ~fill_group,
  colors = c(
    "Male - DHS Official"   = "#2471A3", 
    "Female - DHS Official" = "#7FB3D5", 
    "Male - Imputed ACS"   = "#BA4A00", 
    "Female - Imputed ACS" = "#F5B041"
  ),
  frame = ~year,          
  type = 'bar', 
  text = ~dataset,
  hovertemplate = '<b>%{text}</b><br>%{y:,.0f}<extra></extra>' 
) %>%
  layout(
    barmode = 'stack',    
    
    title = list(
      text = "<b>Interactive Comparison: Imputed ACS vs DHS Official</b>",
      font = list(family = "Arial", size = 16),
      x = 0.5 
    ),
    
    # Map the numeric Plotly axis back to text labels
    xaxis = list(
      title = "<b>Age Bracket</b>",
      tickvals = 1:5,
      ticktext = levels(plot_data$age_bracket),
      tickfont = list(size = 12)
    ),
    
    yaxis = list(
      title = "<b>Total Population Estimate</b>",
      tickformat = ",d", 
      range = c(0, max(
        plot_data %>% 
          group_by(year, plot_x) %>% 
          summarise(total = sum(population_count, na.rm=TRUE), .groups="drop") %>% 
          pull(total)
      ) * 1.1)
    ),
    
    # 1. Push the legend further down (from -0.25 to -0.5 or -0.6)
    legend = list(
      orientation = "h",  
      x = 0.5,            
      xanchor = "center",
      y = -0.5            
    ),
    
    # 2. Increase the bottom margin significantly (from 100 to 160)
    margin = list(b = 160) 
  ) %>%
  animation_opts(
    frame = 1200,         
    transition = 500,     
    redraw = FALSE        
  ) %>%
  animation_slider(
    currentvalue = list(prefix = "<b>Year:</b> ", font = list(color = "black")),
    pad = list(t = 40)
  )

Summary

From these simple comparisons we can notice that our GBM imputations appears to overestimate the number of undocumented immigrants in age brackets 18-24 and 25-34, and underestimate undocumented immigrants in age brackets 35-44 and 45-54.
