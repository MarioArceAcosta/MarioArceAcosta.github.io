top_75_sample_num <- test_gbm_num %>%
  filter(undocu_likely == 1) %>%
  arrange(desc(p)) %>%
  mutate(
    rank = row_number(),
    cumulative_p = rank / n(),
    top_75 = cumulative_p <= 0.75
  ) %>%
  filter(top_75==TRUE)
threshold_75_num <- min(top_75_sample_num$p)

undocu_test_num <- test_gbm_num %>%
  filter(undocu_likely == 1) %>%
  arrange(desc(p))
